     MAKE_JACKET_POTATO,

   public static String getActionDescription(ActionType actionType) {

       case BAKE_ACTION:

       case PLACE_INGREDIENT:
         return "Place Item";
       case MAKE_BURGER:
         return "Make Burger";
       case MAKE_SALAD:
         return "Make Salad";
       case MAKE_PIZZA:
         return "Make Pizza";
       case MAKE_JACKET_POTATO:
         return "Make Jacket Potato";
       case SUBMIT_ORDER:
         return "Submit Order";
       case CLEAR_STATION:
         return "Clear Station";

         return "Unknown Action";
     }
